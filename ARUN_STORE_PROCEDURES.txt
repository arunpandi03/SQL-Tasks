CREATE PROCEDURE ARUNPANDI_MPROCEDURE @menu_id int, @menu_name nvarchar(10)
AS
SELECT * FROM ARUN_MENU WHERE menu_id = @menu_id AND menu_name = @menu_name
SELECT COUNT(menu_id) as count_of_id FROM ARUN_MENU;  
GO

EXEC ARUNPANDI_MPROCEDURE @menu_id=3 , @menu_name ='dosa'

DROP PROCEDURE ARUNPANDI_MPROCEDURE ;
